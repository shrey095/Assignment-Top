Q DISPLAY ALL DATA OF EMPLOYEES TABLE
SQL>select * from employee;

Q DISPLAY FIRST_NAME OF EMPLOYEES TABLE
SQL>select FIRST_NAME from employee;

Q DISPLAY LAST_NAME AND HIRE_DATE OF EMPLOYEES TABLE
SQL>select LAST_NAME,HIRE_DATE from employee; 

Q DISPLAY FIRST_NAME , SALARY AND NEW SALARY (NEW SALARY = SALARY + 1500)
SQL>select FIRST_NAME,SALARY,SALARY+1500 as NEW_SALARY from employee;

Q DISPLAY LAST_NAME , SALARY AND ANNUAL SALARY (ANNUAL SALARY = SALARY * 12)
SQL>select LAST_NAME,SALARY,SALARY*12 as ANNUAL_SALARY from employee;

Q DISPLAY FIRST_NAME , SALARY AND DAILY SALARY (DAILY SALARY = SALARY / 30)
SQL>select FIRST_NAME,SALARY,SALARY/30 as DAILY_SALARY from employee;

Q DISPLAY FIRST_NAME AND LAST_NAME IN SINGLE COLUMN WITH PROPER COLUMN ALIAS/HEADING.
SQL>select FIRST_NAME,LAST_NAME,FIRST_NAME||' '||LAST_NAME as FULL_NAME from employee;
or select FIRST_NAME,LAST_NAME,concat(FIRST_NAME,'  ',LAST_NAME) as FULL_NAME from employee;

Q DISPLAY GIVEN RESULT FOR EACH EMPLOYEE : Steven works as AD_PRES.
SQL>select FIRST_NAME,JOB_ID,FIRST_NAME||"works as"||JOB_ID as FJ from employee;
or select FIRST_NAME,JOB_ID,concat(FIRST_NAME,"works as  ",JOB_ID) as FJ from employee;

13 DISPLAY GIVEN RESULT FOR EACH EMPLOYEE : King's monthly salary is Rs.24000 
SQL>select LAST_NAME,SALARY,concat(LAST_NAME,"  monthly salary is Rs. ",SALARY) as LS from employee;

Q DISPLAY ALL INFO. OF EMPLOYEES TABLE WITH ANNUAL SALARY
SQL> select *,(SALARY*12) as ANNUAL_SALARY from employee;

Q DISPLAY FIRST_NAME , NEW_ANNUAL_SALARY (NEW_ANNUAL_SALARY = SALARY + 1000 * 12)
SQL>select FIRST_NAME,((SALARY + 1000)*12) as NEW_ANNUAL_SALARY from employee;
